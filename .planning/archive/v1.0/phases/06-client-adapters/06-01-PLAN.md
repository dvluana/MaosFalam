<!-- API contracts the executor needs. Read from existing route files. -->

POST /api/lead/register
Body: { name, email, gender: "female"|"male", session_id, email_opt_in: boolean }
Returns 201: { lead_id: string }
Returns 400: { error: string } on validation failure

POST /api/reading/capture
Body: { photo_base64, session_id, lead_id, target_name, target_gender, is_self }
Returns 200: { reading_id: string, report: ReportJSON }
Returns 422: { error: string, code: "LOW_CONFIDENCE" }
Returns 429: rate limited

GET /api/reading/[id]
Returns 200: { reading: { id, target_name, target_gender, tier, report: ReportJSON, created_at } }
Returns 404: reading not found
Returns 410: reading deactivated

```typescript
export type HandElement = "fire" | "water" | "earth" | "air";
export type Tier = "free" | "premium";

export interface ReportJSON {
  opening: string;
  impact_phrase: string;
  element: { key: HandElement; intro: string; body: string };
  portrait: {
    dominant_mount: string;
    lines_detected: string;
    mounts_mapped: string;
    rare_signs_found: string;
    exclusivity: string;
  };
  sections: ReportSection[];
  venus: ReportVenus;
  crossings: ReportCrossing[];
  compatibility: ReportCompat[];
  rare_signs: ReportRareSign[];
  epilogue: string;
}

export interface Reading {
  id: string;
  tier: Tier;
  share_token: string;
  share_expires_at: string;
  report: ReportJSON;
  created_at: string;
}
```

Note on GET /api/reading/[id] response:
The route returns { reading: { id, target_name, target_gender, tier, report, created_at } }
which does NOT include share_token or share_expires_at (those are not in the DB query).
The Reading type requires share_token — use reading.id as share_token fallback since
the share URL in compartilhar/[token] will also use the reading id directly.
