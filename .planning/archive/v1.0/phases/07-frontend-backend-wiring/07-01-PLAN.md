    Exact implementation:
    ```typescript
    "use client";

    import { useClerk, useUser } from "@clerk/nextjs";
    import { useCallback } from "react";

    interface AuthUser {
      id: string;
      name: string;
      email: string;
    }

    export function useAuth() {
      const { user: clerkUser, isLoaded, isSignedIn } = useUser();
      const { signOut } = useClerk();

      const user: AuthUser | null =
        isLoaded && isSignedIn && clerkUser
          ? {
              id: clerkUser.id,
              name: clerkUser.fullName ?? clerkUser.primaryEmailAddress?.emailAddress ?? "",
              email: clerkUser.primaryEmailAddress?.emailAddress ?? "",
            }
          : null;

      const login = useCallback((_email: string, _password: string): boolean => false, []);

      const register = useCallback(
        (_name: string, _email: string, _password: string): boolean => false,
        [],
      );

      const logout = useCallback((): void => {
        void signOut();
      }, [signOut]);

      return { user, hydrated: isLoaded, login, register, logout };
    }
    ```

    Note: `login` and `register` return `false` intentionally — login/registro pages rely on Clerk's hosted flow via ClerkProvider; these mock functions are no longer the auth path. The pages may call them but nothing breaks since the Clerk SignIn component handles the real flow.
