    1. In src/lib/mediapipe.ts, add new exported function computeElementHint:

    ```typescript
    type HandElement = "fire" | "water" | "earth" | "air";

    export function computeElementHint(landmarks: NormalizedLandmark[]): HandElement | undefined {
      if (landmarks.length < 21) return undefined;
      const palmWidth = dist(landmarks[5], landmarks[17]); // INDEX_MCP to PINKY_MCP
      const palmHeight = dist(landmarks[0], landmarks[9]); // WRIST to MIDDLE_MCP
      const fingerLength = dist(landmarks[9], landmarks[12]); // MIDDLE_MCP to MIDDLE_TIP
      if (palmWidth < 0.01 || palmHeight < 0.01) return undefined;
      const isLongPalm = palmHeight / palmWidth > 1.1;
      const isLongFingers = fingerLength / palmHeight > 0.95;
      if (!isLongPalm && isLongFingers) return "air";
      if (!isLongPalm && !isLongFingers) return "earth";
      if (isLongPalm && !isLongFingers) return "fire";
      return "water";
    }
    ```

    NOTE: The `dist` function already exists in mediapipe.ts (line 50). It is currently not exported. computeElementHint uses it internally -- no need to export dist.

    - MIDDLE_TIP = 12
