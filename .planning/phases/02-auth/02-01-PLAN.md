```typescript
import { clerkMiddleware, createRouteMatcher } from "@clerk/nextjs/server";

const isProtectedRoute = createRouteMatcher([
  "/api/reading/new(.*)",
  "/api/credits/(.*)",
  "/api/user/(.*)",
  "/conta(.*)",
]);

export default clerkMiddleware(async (auth, req) => {
  if (isProtectedRoute(req)) {
    await auth.protect();
  }
});

export const config = {
  matcher: [
    "/((?!_next|[^?]*\\.(?:html?|css|js(?!on)|jpe?g|png|webp|gif|svg|ttf|woff2?|ico|csv|docx?|xlsx?|zip|webmanifest)).*)",
    "/(api|trpc)(.*)",
  ],
};
```

Note: the config.matcher is UPDATED vs middleware.ts — two patterns instead of one:

1. Static asset exclusion pattern (same concept, adds `(?!on)` to js exclusion to allow json)
2. Explicit API/trpc catch-all to ensure all API routes are evaluated

This is the standard Clerk recommended matcher for Next.js App Router.

IMPORTANT: Do NOT include any manifesto rewrite logic here — proxy.ts for this project
is solely for Clerk auth. The manifesto route is handled elsewhere.
