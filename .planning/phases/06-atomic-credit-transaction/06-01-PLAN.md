       ```typescript
       import { prisma } from "@/server/lib/prisma";
       import { logger } from "@/server/lib/logger";

       interface DebitResult {
         debited: boolean;
         packId?: string;
       }

       export async function debitCreditFIFO(clerkUserId: string): Promise<DebitResult> {
         // Find oldest pack with remaining > 0
         const pack = await prisma.creditPack.findFirst({
           where: { clerkUserId, remaining: { gt: 0 } },
           orderBy: { createdAt: "asc" },
           select: { id: true },
         });

         if (!pack) return { debited: false };

         // Atomic decrement — returns number of rows affected
         // If another request already decremented to 0, this returns 0 rows
         const affected: [{ cnt: bigint }] = await prisma.$queryRawUnsafe(
           `UPDATE credit_packs SET remaining = remaining - 1 WHERE id = $1 AND remaining > 0 RETURNING id`,
           pack.id,
         );

         if (affected.length === 0) {
           // Race condition: pack was decremented by another request between findFirst and UPDATE
           // Retry once with next available pack
           logger.info({ clerkUserId }, "Credit debit race detected, retrying");
           return debitCreditFIFO(clerkUserId);
         }

         return { debited: true, packId: pack.id };
       }
       ```
       Note: Use `$queryRawUnsafe` with parameterized values ($1) for SQL injection safety. The RETURNING clause lets us check affected rows. Add recursion guard (max 3 retries) to prevent infinite loop.
